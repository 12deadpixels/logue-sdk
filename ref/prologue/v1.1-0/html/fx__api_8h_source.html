133  __fast_inline float fx_sinf(float x) {
134  const float p = x - (uint32_t)x;
135 
136  // half period stored -- wrap around and invert
137  const float x0f = 2.f * p * k_wt_sine_size;
138  const uint32_t x0p = (uint32_t)x0f;
139 
140  const uint32_t x0 = x0p & k_wt_sine_mask;
141  const uint32_t x1 = (x0 + 1) & k_wt_sine_mask;
142 
143  const float y0 = linintf(x0f - x0p, wt_sine_lut_f[x0], wt_sine_lut_f[x1]);
144  return (x0p < k_wt_sine_size)?y0:-y0;
145  }
